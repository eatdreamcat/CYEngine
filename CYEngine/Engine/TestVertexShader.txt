#version 330 core
layout(location = 8) in vec3 aPos;
layout(location = 9) in vec4 aColor;
layout(location = 9) in vec2 aTexCoord;
uniform vec4 ourColor;
uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;
out vec4 vertexColor;
out vec2 TexCoord;
void main() {
    gl_Position = projection*view*model*vec4(aPos, 1.0);
    vertexColor = aColor;
    TexCoord = aTexCoord;
}
